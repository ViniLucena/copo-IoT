1) consumo do cliente ao longo do tempo
query:
SELECT
 time_bucket('60 seconds', data) AS time,
 AVG(hc.quantidade_ml) AS media_volume,
 AVG(hc.temperatura_c) AS media_temperatura
FROM copos
INNER JOIN historico_copo hc ON hc.id_copo = copos.id_copo
WHERE mesa = 1 AND data > NOW() - INTERVAL '1 day'
GROUP BY time
ORDER BY time ASC

2) quando tempo cada copo fica vazio

query:

SELECT
    CONCAT(cl.nome_completo, ' (Copo ', c.id, ')') AS cliente,
    ROUND(
        EXTRACT(EPOCH FROM (NOW() - c.vazio_a)) / 60,
        1
    ) AS minutos_vazio
FROM copos c

JOIN vinculo_copo_cliente vcc
    ON c.id = vcc.id_copo

JOIN clientes cl
    ON vcc.id_cliente = cl.id

WHERE c.mesa = 1
  AND c.quantidade_ml = 0
  AND c.vazio_a IS NOT NULL
  AND vcc.ativo = true

ORDER BY minutos_vazio DESC;

3) Velocidade média de consumo da mesa (mL/min)
query:
WITH consumo_2_dias AS (
    SELECT
        cb.id_copo,

        FIRST_VALUE(cb.quantidade_ml) OVER (
            PARTITION BY cb.id_copo
            ORDER BY cb.created_at ASC
        ) AS volume_inicial,

        LAST_VALUE(cb.quantidade_ml) OVER (
            PARTITION BY cb.id_copo
            ORDER BY cb.created_at ASC
            ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
        ) AS volume_atual

    FROM consumo_bebidas cb

    JOIN copos c
        ON cb.id_copo = c.id

    WHERE c.mesa = 1
      AND cb.created_at >= NOW() - INTERVAL '2 days'
),

velocidade_por_copo AS (
    SELECT DISTINCT
        id_copo,
        (volume_inicial - volume_atual) AS ml_consumidos
    FROM consumo_2_dias
)

SELECT
    COALESCE(
        SUM(ml_consumidos) / (2 * 24 * 60.0),
        0
    ) AS "Velocidade Média da Mesa (ml/min)"
FROM velocidade_por_copo;

4) Temperatura média dos copos da mesa (°C)
SELECT
    COALESCE(AVG(temperatura_c), 0) AS "Temperatura Média da Mesa"
FROM copos
WHERE mesa = 1
  AND temperatura_c IS NOT NULL;
